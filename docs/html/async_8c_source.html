242static void* worker_thread_func(void* arg) {
243 AsyncEventLoop* loop = (AsyncEventLoop*)arg;
244
245 while (1) {
246 // Get a task from the queue
247 AsyncTaskNode* node = NULL;
248
249 pthread_mutex_lock(&loop->queue_mtx);
250
251 // Wait until there's a task or the loop is stopping
252 while (loop->queue_head == NULL && loop->running) {
253 pthread_cond_wait(&loop->queue_cv, &loop->queue_mtx);
254 }
255
256 // Check if we should exit
257 if (!loop->running && loop->queue_head == NULL) {
258 pthread_mutex_unlock(&loop->queue_mtx);
259 break;
260 }
261
262 // Dequeue a task
263 if (loop->queue_head) {
264 node = loop->queue_head;
265 loop->queue_head = node->next;
266
267 // Update tail if queue is now empty
268 if (loop->queue_head == NULL) {
269 loop->queue_tail = NULL;
270 }
271 }
272
273 pthread_mutex_unlock(&loop->queue_mtx);
274
275 // Process the task
276 if (node) {
277 AsyncTask* task = node->task;
278
279 // Check for cancellation
280 int cancelled = 0;
281 pthread_mutex_lock(&task->mtx);
282 cancelled = task->cancel;
283 pthread_mutex_unlock(&task->mtx);
284
285 if (!cancelled) {
286 // Execute the function
287 Result result = node->func->func(node->params);
288
289 // Update task with result
290 pthread_mutex_lock(&task->mtx);
291 task->result = result;
292 task->ready = 1;
293 pthread_cond_broadcast(&task->cv);
294 pthread_mutex_unlock(&task->mtx);
295 } else {
296 // Task was cancelled, mark as ready without executing
297 pthread_mutex_lock(&task->mtx);
298 task->ready = 1;
299 pthread_cond_broadcast(&task->cv);
300 pthread_mutex_unlock(&task->mtx);
301 }
302
303 // Free the task node (but not the task itself, which is freed after await)
304 free(node);
305 }
306 }
307
