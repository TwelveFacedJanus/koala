184 inline void event_loop_worker(AsyncEventLoop* loop) {
185 while (true) {
186 AsyncTaskNode* node = nullptr;
187 {
188 std::unique_lock<std::mutex> lock(loop->queue_mtx);
189 loop->queue_cv.wait(lock, [&]{ return !loop->queue.empty() || !loop->running; });
190 if (!loop->running && loop->queue.empty()) break;
191 if (!loop->queue.empty()) {
192 node = loop->queue.front();
193 loop->queue.pop();
194 }
195 }
196 if (node) {
197 if (node->task->cancel) {
198 // Отмена задачи
199 std::lock_guard<std::mutex> lock(node->task->mtx);
200 node->task->ready = true;
201 } else {
202 Result res = node->func->func(node->params);
203 {
204 std::lock_guard<std::mutex> lock(node->task->mtx);
205 node->task->result = res;
206 node->task->ready = true;
207 }
208 }
209 node->task->cv.notify_all();
210 delete node;
211 }
212 }
213 }
214}

226inline void async_event_loop_stop(AsyncEventLoop* loop) {
227 {
228 std::lock_guard<std::mutex> lock(loop->queue_mtx);
229 loop->running = false;
230 }
231 loop->queue_cv.notify_all();
232 for (auto& worker : loop->workers) {
233 if (worker && worker->joinable()) worker->join();
234 delete worker;
235 }
236 loop->workers.clear();
237 delete loop;
